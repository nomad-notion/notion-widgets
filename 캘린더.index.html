<script>

    // JS: 기능 영역

    const date = new Date();

    

    // 1. 왼쪽 패널 데이터 채우기

    const days = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];

    document.getElementById("weekday").innerText = days[date.getDay()];

    document.getElementById("dayNumber").innerText = date.getDate();


    // 2. 오른쪽 달력 그리기

    const months = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];

    document.getElementById("monthName").innerText = months[date.getMonth()];


    // 달력 로직 (간소화) - 실제로는 해당 월의 시작 요일 등을 계산해야 함

    // 예시로 1~31일까지 숫자를 뿌려줍니다.

    const grid = document.getElementById("calendarGrid");

    

    // 요일 헤더 (M T W T F S S)

    const headers = ["S", "M", "T", "W", "T", "F", "S"];

    headers.forEach(h => {

        const div = document.createElement("div");

        div.className = "calendar-header";

        div.innerText = h;

        grid.appendChild(div);

    });


    // 날짜 채우기 (오늘 날짜 표시 포함)

    const today = date.getDate();

    // 간단한 예시를 위해 1일부터 30/31일까지 반복

    for (let i = 1; i <= 31; i++) {

        const div = document.createElement("div");

        div.innerText = i;

        if (i === today) {

            div.className = "today-circle"; // 오늘 날짜 동그라미

        }

        grid.appendChild(div);

    }

</script>
